# CLAUDE.md Auto-Sync Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** `fubbik sync-claude-md` watches chunks tagged "claude-context" (or in a named collection) and generates/updates a `.claude/CLAUDE.md` file, keeping AI assistant context fresh.

**Architecture:** New CLI command that fetches chunks matching a tag or collection, formats them using the existing context export scoring algorithm, and writes to a file. Optionally runs in watch mode (polls for changes). Also add an API endpoint so the web UI can trigger a preview.

**Tech Stack:** Commander.js, Bun, fetch API, existing context export service

// packages/api/src/context-export/claude-md.ts
import { Effect } from "effect";
import { listChunks } from "@fubbik/db/repository";

interface ClaudeMdOptions {
    userId: string;
    codebaseId?: string;
    tag?: string;           // default: "claude-context"
    maxTokens?: number;     // default: 8000
    codebaseName?: string;
}

export function generateClaudeMd(opts: ClaudeMdOptions) {
    return Effect.gen(function* () {
        const tag = opts.tag ?? "claude-context";

        // IMPORTANT: The `tags` param in listChunks at the repo level is NOT actually
        // implemented (the filter is stubbed out). You must either:
        // (a) Implement tag filtering in listChunks by joining chunk_tag table, OR
        // (b) Fetch all chunks then filter by tag client-side using getTagsForChunk
        //
        // Simplest approach: fetch all chunks, then for each check if it has the target tag.
        // Use getTagsForChunk from @fubbik/db/repository for each chunk.
        // Or better: write a direct SQL query joining chunk + chunk_tag + tag tables.
        //
        // For now, fetch all and filter:
        const allResult = yield* listChunks({
            userId: opts.userId,
            codebaseId: opts.codebaseId,
            sort: "updated",
            limit: 200,
            offset: 0,
        });

        // Filter by tag — read getTagsForChunk or query chunk_tag join directly
        // This is a simplification; a proper implementation should do the join at DB level
        const result = { chunks: allResult.chunks }; // TODO: filter by tag at DB level

        if (result.chunks.length === 0) {
            return { content: "", chunks: 0 };
        }

        // Group chunks by type for structured output
        const conventions = result.chunks.filter(c => c.type === "note");
        const documents = result.chunks.filter(c => c.type === "document");
        const references = result.chunks.filter(c => c.type === "reference");
        const other = result.chunks.filter(c => !["note", "document", "reference"].includes(c.type));

        // Build markdown
        const sections: string[] = [];
        sections.push(`# ${opts.codebaseName ?? "Project"} Knowledge Context`);
        sections.push("");
        sections.push(`> Auto-generated by fubbik. ${result.chunks.length} chunks tagged \`${tag}\`.`);
        sections.push(`> Last updated: ${new Date().toISOString()}`);
        sections.push("");

        const addSection = (title: string, chunks: typeof conventions) => {
            if (chunks.length === 0) return;
            sections.push(`## ${title}`);
            sections.push("");
            for (const c of chunks) {
                sections.push(`### ${c.title}`);
                sections.push("");
                sections.push(c.content);
                if (c.rationale) {
                    sections.push("");
                    sections.push(`**Rationale:** ${c.rationale}`);
                }
                sections.push("");
            }
        };

        addSection("Conventions", conventions);
        addSection("Architecture", documents);
        addSection("References", references);
        addSection("Other", other);

        return { content: sections.join("\n"), chunks: result.chunks.length };
    });
}
```

// apps/cli/src/commands/sync-claude-md.ts
import { Command } from "commander";
import { getServerUrl } from "../lib/store";
import { outputError } from "../lib/output";
import { formatSuccess, formatDim } from "../lib/colors";
import { writeFileSync, mkdirSync, existsSync } from "fs";
import { join, dirname } from "path";

export const syncClaudeMdCommand = new Command("sync-claude-md")
    .description("Generate .claude/CLAUDE.md from tagged chunks")
    .option("--tag <tag>", "tag to filter by", "claude-context")
    .option("--output <path>", "output file path", ".claude/CLAUDE.md")
    .option("--codebase <name>", "scope to codebase")
    .option("--watch", "poll for changes every 60s")
    .option("--interval <seconds>", "poll interval in seconds", "60")
    .action(async (opts) => {
        let serverUrl: string;
        try { serverUrl = getServerUrl()!; if (!serverUrl) throw new Error(); }
        catch { outputError("Server URL required. Run 'fubbik init'."); return; }

        const generate = async () => {
            const params = new URLSearchParams({ tag: opts.tag });
            if (opts.codebase) params.set("codebaseId", opts.codebase);

            const res = await fetch(`${serverUrl}/api/chunks/export/claude-md?${params}`);
            if (!res.ok) { outputError(`API error: ${res.statusText}`); return false; }

            const data = await res.json() as { content: string; chunks: number };
            if (data.chunks === 0) {
                console.error(formatDim(`No chunks tagged "${opts.tag}" found. Tag chunks to include them.`));
                return false;
            }

            const outPath = opts.output;
            const dir = dirname(outPath);
            if (!existsSync(dir)) mkdirSync(dir, { recursive: true });
            writeFileSync(outPath, data.content);
            console.error(formatSuccess(`Written ${data.chunks} chunks to ${outPath}`));
            return true;
        };

        await generate();

        if (opts.watch) {
            const interval = Number(opts.interval) * 1000;
            console.error(formatDim(`Watching for changes every ${opts.interval}s...`));
            setInterval(generate, interval);
        }
    });
```

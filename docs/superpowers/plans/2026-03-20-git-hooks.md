# Chunk-Aware Git Hooks Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A CLI command that installs a git pre-commit hook warning developers when modified files match chunk `appliesTo` globs or `fileReferences`, prompting them to review related knowledge.

**Architecture:** New CLI command `fubbik hooks install` that writes a shell script to `.git/hooks/pre-commit`. The hook runs `fubbik check-files` (new command) which takes a list of staged files, queries the server for matching chunks via appliesTo globs and fileRef lookups, and prints warnings. No blocking — just informational output.

**Tech Stack:** Commander.js, Bun, shell scripting, fubbik API (file-refs/lookup, applies-to)

```ts
// apps/cli/src/lib/glob-match.ts

/**
 * Simple glob match — supports *, **, and ? patterns.
 * No external dependencies.
 */
export function globMatch(pattern: string, path: string): boolean {
    const regexStr = pattern
        .replace(/\./g, "\\.")
        .replace(/\*\*/g, "{{GLOBSTAR}}")
        .replace(/\*/g, "[^/]*")
        .replace(/\?/g, "[^/]")
        .replace(/\{\{GLOBSTAR\}\}/g, ".*");
    return new RegExp(`^${regexStr}$`).test(path);
}
```

```ts
// apps/cli/src/commands/check-files.ts
import { Command } from "commander";
import { getServerUrl } from "../lib/store";
import { globMatch } from "../lib/glob-match";
import { formatSuccess, formatError, formatBold, formatDim, formatType } from "../lib/colors";
import { output, outputError, isJson } from "../lib/output";

export const checkFilesCommand = new Command("check-files")
    .description("Check if staged files match any chunk appliesTo or fileReferences")
    .argument("[files...]", "file paths to check (defaults to git staged files)")
    .option("--staged", "automatically detect staged files via git")
    .action(async (files: string[], opts: { staged?: boolean }, cmd: Command) => {
        let serverUrl: string;
        try {
            serverUrl = getServerUrl()!;
            if (!serverUrl) throw new Error();
        } catch {
            // No server configured — skip silently in hooks context
            return;
        }

        // Get files to check
        let filesToCheck = files;
        if (opts.staged || filesToCheck.length === 0) {
            const { execSync } = await import("child_process");
            try {
                const staged = execSync("git diff --cached --name-only", { encoding: "utf-8" });
                filesToCheck = staged.trim().split("\n").filter(Boolean);
            } catch {
                return; // Not in a git repo
            }
        }

        if (filesToCheck.length === 0) return;

        // Fetch all chunks with appliesTo patterns
        const chunksRes = await fetch(`${serverUrl}/api/chunks?limit=100`);
        if (!chunksRes.ok) return;
        const { chunks } = await chunksRes.json();

        // Fetch appliesTo patterns for all chunks
        const matchedChunks: Array<{ id: string; title: string; type: string; matchedFile: string; matchType: string }> = [];

        for (const file of filesToCheck) {
            // Check file-refs lookup
            const refRes = await fetch(`${serverUrl}/api/file-refs/lookup?path=${encodeURIComponent(file)}`);
            if (refRes.ok) {
                const refs = await refRes.json();
                for (const ref of refs) {
                    matchedChunks.push({
                        id: ref.id,
                        title: ref.title,
                        type: ref.type || "note",
                        matchedFile: file,
                        matchType: "file-ref",
                    });
                }
            }
        }

        // Check appliesTo globs (need to fetch per chunk)
        // For efficiency, fetch chunks that have appliesTo patterns
        for (const c of chunks) {
            const atRes = await fetch(`${serverUrl}/api/chunks/${c.id}/applies-to`);
            if (!atRes.ok) continue;
            const patterns = await atRes.json();
            for (const pat of patterns) {
                for (const file of filesToCheck) {
                    if (globMatch(pat.pattern, file)) {
                        if (!matchedChunks.find(m => m.id === c.id && m.matchedFile === file)) {
                            matchedChunks.push({
                                id: c.id,
                                title: c.title,
                                type: c.type,
                                matchedFile: file,
                                matchType: "applies-to",
                            });
                        }
                    }
                }
            }
        }

        if (matchedChunks.length === 0) return;

        if (isJson(cmd)) {
            console.log(JSON.stringify(matchedChunks, null, 2));
            return;
        }

        // Print warnings
        console.error("");
        console.error(formatBold("⚠ Knowledge chunks may need updating:"));
        console.error("");

        const grouped = new Map<string, typeof matchedChunks>();
        for (const m of matchedChunks) {
            const key = `${m.id}:${m.title}`;
            if (!grouped.has(key)) grouped.set(key, []);
            grouped.get(key)!.push(m);
        }

        for (const [, matches] of grouped) {
            const first = matches[0]!;
            console.error(`  ${formatType(first.type)} ${formatBold(first.title)} ${formatDim(`(${first.id.slice(0, 8)})`)}`);
            for (const m of matches) {
                console.error(`    ${formatDim("→")} ${m.matchedFile} ${formatDim(`(${m.matchType})`)}`);
            }
        }

        console.error("");
        console.error(formatDim(`Review: ${serverUrl}/chunks`));
        console.error("");
    });
```

```ts
// apps/cli/src/commands/hooks.ts
import { Command } from "commander";
import { formatSuccess, formatError } from "../lib/colors";

const HOOK_SCRIPT = `#!/bin/sh
# fubbik pre-commit hook — warns about chunks affected by staged changes
# Installed by: fubbik hooks install

# Only warn, never block
fubbik check-files --staged 2>&1 || true
`;

export const hooksCommand = new Command("hooks")
    .description("Manage git hooks")
    .addCommand(
        new Command("install")
            .description("Install pre-commit hook that warns about affected chunks")
            .option("--force", "overwrite existing hook")
            .action(async (opts: { force?: boolean }) => {
                const { existsSync, writeFileSync, chmodSync, readFileSync } = await import("fs");
                const { join } = await import("path");
                const { execSync } = await import("child_process");

                // Find git root
                let gitRoot: string;
                try {
                    gitRoot = execSync("git rev-parse --show-toplevel", { encoding: "utf-8" }).trim();
                } catch {
                    console.error(formatError("Not in a git repository"));
                    return;
                }

                const hookPath = join(gitRoot, ".git", "hooks", "pre-commit");

                // Check for existing hook
                if (existsSync(hookPath) && !opts.force) {
                    const existing = readFileSync(hookPath, "utf-8");
                    if (existing.includes("fubbik")) {
                        console.error(formatSuccess("Hook already installed"));
                        return;
                    }
                    console.error(formatError("A pre-commit hook already exists. Use --force to overwrite."));
                    return;
                }

                writeFileSync(hookPath, HOOK_SCRIPT);
                chmodSync(hookPath, 0o755);
                console.error(formatSuccess("Pre-commit hook installed"));
            })
    )
    .addCommand(
        new Command("uninstall")
            .description("Remove fubbik pre-commit hook")
            .action(async () => {
                const { existsSync, readFileSync, unlinkSync } = await import("fs");
                const { join } = await import("path");
                const { execSync } = await import("child_process");

                let gitRoot: string;
                try {
                    gitRoot = execSync("git rev-parse --show-toplevel", { encoding: "utf-8" }).trim();
                } catch {
                    console.error(formatError("Not in a git repository"));
                    return;
                }

                const hookPath = join(gitRoot, ".git", "hooks", "pre-commit");
                if (!existsSync(hookPath)) {
                    console.error("No pre-commit hook found.");
                    return;
                }

                const content = readFileSync(hookPath, "utf-8");
                if (!content.includes("fubbik")) {
                    console.error("Pre-commit hook exists but was not installed by fubbik.");
                    return;
                }

                unlinkSync(hookPath);
                console.error(formatSuccess("Pre-commit hook removed"));
            })
    );
```

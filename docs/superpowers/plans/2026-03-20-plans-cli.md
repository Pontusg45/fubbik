```ts
// apps/cli/src/commands/plan.ts
import { Command } from "commander";
import { getServerUrl } from "../lib/store";
import { outputError, isJson } from "../lib/output";
import { formatSuccess, formatError, formatBold, formatDim, formatType } from "../lib/colors";

async function fetchApi(path: string, opts?: RequestInit) {
    // NOTE: getServerUrl() throws if no store file exists (not returns undefined).
    // The try/catch in each action handler catches this.
    const serverUrl = getServerUrl();
    if (!serverUrl) throw new Error("No server URL configured. Run 'fubbik init' first.");
    const res = await fetch(`${serverUrl}/api${path}`, {
        ...opts,
        headers: { "Content-Type": "application/json", ...opts?.headers },
    });
    if (!res.ok) {
        const body = await res.json().catch(() => ({}));
        throw new Error(body.message || `API error: ${res.status}`);
    }
    return res.json();
}

export const planCommand = new Command("plan")
    .description("Manage implementation plans")

    // ─── plan create ───
    .addCommand(
        new Command("create")
            .description("Create a new plan")
            .requiredOption("-t, --title <title>", "plan title")
            .option("-d, --description <desc>", "plan description")
            .option("--steps <steps...>", "initial step descriptions")
            .option("--codebase <name>", "scope to codebase")
            .action(async (opts, cmd) => {
                try {
                    const body: Record<string, unknown> = { title: opts.title };
                    if (opts.description) body.description = opts.description;
                    if (opts.steps) {
                        body.steps = opts.steps.map((s: string, i: number) => ({
                            description: s,
                            order: i,
                        }));
                    }
                    const plan = await fetchApi("/plans", { method: "POST", body: JSON.stringify(body) });
                    if (isJson(cmd)) { console.log(JSON.stringify(plan, null, 2)); return; }
                    console.error(formatSuccess(`Created plan: ${formatBold(plan.title)}`));
                    console.error(formatDim(`  ID: ${plan.id}`));
                    if (plan.steps?.length) {
                        console.error(formatDim(`  ${plan.steps.length} step(s) added`));
                    }
                } catch (e: any) { outputError(e.message); }
            })
    )

    // ─── plan list ───
    .addCommand(
        new Command("list")
            .description("List plans")
            .option("--status <status>", "filter by status (draft/active/completed/archived)")
            .action(async (opts, cmd) => {
                try {
                    const params = new URLSearchParams();
                    if (opts.status) params.set("status", opts.status);
                    const plans = await fetchApi(`/plans?${params}`);
                    if (isJson(cmd)) { console.log(JSON.stringify(plans, null, 2)); return; }
                    if (plans.length === 0) { console.error("No plans found."); return; }
                    for (const p of plans) {
                        const pct = p.progress ?? 0;
                        const bar = "█".repeat(Math.round(pct / 5)) + "░".repeat(20 - Math.round(pct / 5));
                        console.error(`  ${formatBold(p.title)} ${formatDim(`(${p.id.slice(0, 8)})`)}`);
                        console.error(`    ${formatType(p.status)} ${bar} ${pct}% (${p.doneCount}/${p.stepCount})`);
                    }
                } catch (e: any) { outputError(e.message); }
            })
    )

    // ─── plan show ───
    .addCommand(
        new Command("show")
            .description("Show plan details with steps")
            .argument("<id>", "plan ID")
            .action(async (id, _opts, cmd) => {
                try {
                    const plan = await fetchApi(`/plans/${id}`);
                    if (isJson(cmd)) { console.log(JSON.stringify(plan, null, 2)); return; }
                    console.error(formatBold(plan.title));
                    console.error(`${formatType(plan.status)} ${plan.progress}% complete`);
                    if (plan.description) console.error(formatDim(plan.description));
                    console.error("");
                    for (const step of plan.steps ?? []) {
                        const icon = step.status === "done" ? "✓" : step.status === "in_progress" ? "→" : step.status === "blocked" ? "✗" : "○";
                        const color = step.status === "done" ? formatSuccess : step.status === "blocked" ? formatError : (s: string) => s;
                        console.error(`  ${color(`${icon} ${step.description}`)}`);
                        if (step.note) console.error(formatDim(`    ${step.note}`));
                    }
                } catch (e: any) { outputError(e.message); }
            })
    )

    // ─── plan step-done ───
    .addCommand(
        new Command("step-done")
            .description("Mark a step as done")
            .argument("<planId>", "plan ID")
            .argument("<stepId>", "step ID")
            .option("-n, --note <note>", "add a note to the step")
            .action(async (planId, stepId, opts, cmd) => {
                try {
                    const body: Record<string, unknown> = { status: "done" };
                    if (opts.note) body.note = opts.note;
                    await fetchApi(`/plans/${planId}/steps/${stepId}`, { method: "PATCH", body: JSON.stringify(body) });
                    if (!isJson(cmd)) console.error(formatSuccess("Step marked as done"));
                } catch (e: any) { outputError(e.message); }
            })
    )

    // ─── plan add-step ───
    .addCommand(
        new Command("add-step")
            .description("Add a step to a plan")
            .argument("<planId>", "plan ID")
            .argument("<description>", "step description")
            .action(async (planId, description, _opts, cmd) => {
                try {
                    const step = await fetchApi(`/plans/${planId}/steps`, { method: "POST", body: JSON.stringify({ description }) });
                    if (isJson(cmd)) { console.log(JSON.stringify(step, null, 2)); return; }
                    console.error(formatSuccess(`Added step: ${description}`));
                } catch (e: any) { outputError(e.message); }
            })
    )

    // ─── plan activate / complete / archive ───
    .addCommand(
        new Command("activate")
            .description("Set plan status to active")
            .argument("<id>", "plan ID")
            .action(async (id, _opts, cmd) => {
                try {
                    await fetchApi(`/plans/${id}`, { method: "PATCH", body: JSON.stringify({ status: "active" }) });
                    if (!isJson(cmd)) console.error(formatSuccess("Plan activated"));
                } catch (e: any) { outputError(e.message); }
            })
    )
    .addCommand(
        new Command("complete")
            .description("Set plan status to completed")
            .argument("<id>", "plan ID")
            .action(async (id, _opts, cmd) => {
                try {
                    await fetchApi(`/plans/${id}`, { method: "PATCH", body: JSON.stringify({ status: "completed" }) });
                    if (!isJson(cmd)) console.error(formatSuccess("Plan completed"));
                } catch (e: any) { outputError(e.message); }
            })
    );
```
